#include 
#include 
int main (void)
{
    float valor;
    int conta;
    float saldo = 20;
    float chequeEspecial = 100;
    int opcao = 1;
    int i;
    float listaClientes[10];
    listaClientes[0] = 100;
    listaClientes[1] = 500;
    listaClientes[2] = 1000;
    listaClientes[3] = 450;
    listaClientes[4] = 600;

    while (opcao > 0)
    {
        // menu
        printf("\n\n***** CAIXA ELETRONICO *****");
        printf("\n* Escolha uma opcao:");
        printf("\n* 1-Saque");
        printf("\n* 2-Deposito");
        printf("\n* 0-Desligar");
        printf("\n* Digite => ");
        scanf("%d", &opcao);

        if(opcao > 2){
            printf("\nOpção não suportada!");
        } else if(opcao == 2) {
            printf("\nEstamos trabalhando na funcionalidade de deposito, aguarde!");
        } else if(opcao == 1) {
            //funcionalidade de saque
            printf("\nInforme sua conta (1-10)...\n");
            scanf("%d", &conta);
            printf("\nSaldo atual da sua conta:");
            printf("%.2f\n", listaClientes[conta]);
            printf("\nInforme o valor do saque...\n");
            scanf("%f", &valor);
            printf("Processando...\n");
            for ( i = 5; i >= 1 ; i--) {
                printf("%d...", i);
                sleep(1);
            }
            if (valor <= listaClientes[conta]) {
                listaClientes[conta] = listaClientes[conta] - valor;
                printf("Saque realizado.\n");
                printf("Saldo: %.2f\n", listaClientes[conta]);
            } else if (valor <= listaClientes[conta] + chequeEspecial){
                //deduzir do saldo
                listaClientes[conta] = listaClientes[conta] - valor;
                //deduzir do cheque especial o valor faltante
                chequeEspecial = chequeEspecial + listaClientes[conta]; //saldo está negativo
                //zerar o saldo
                listaClientes[conta] = listaClientes[conta] + (listaClientes[conta] * -1);
                printf("Saque realizado com cheque especial!\n");
                printf("Saldo: %.2f\n", listaClientes[conta]);
                printf("Cheque especial: %.2f\n", chequeEspecial);
            } else {
                printf("Sem saldo suficiente.\n");
                return 0;
            }
        } else {
            // sair do sistema
            printf("Encerrando...");
            return 0;
        }
    }
}
