#include 

int soma (int valorA, int valorB)
{
    int somatorio;
    somatorio = valorA + valorB;
    return somatorio;
}

int subtracao (int valorA, int valorB)
{
    int resultado;
    resultado = valorA - valorB;
    return resultado;
}

int multiplicacao (int valorA, int valorB)
{
    int resultado;
    resultado = valorA * valorB;
    return resultado;
}

float divisao (int valorA, int valorB)
{
    float resultado;
    resultado = valorA / valorB;
    return resultado;
}

float porcentagem (int total, int porcentagemDesejada)
{
    float resultado;
    resultado = total / 100 * porcentagemDesejada;
    return resultado;
}

int main (void) {
    int numeroA;
    int numeroB;
    int resultado;
    int operacao = 1;
    printf("**** CALCULADORA ****");

    while(operacao != 0)
    {
        printf("\n\nOperacao:\n");
        printf("1-Soma\n2-Subtracao\n3-Multiplicacao\n4-Divisao\n5-Porcentagem\n0-Sair\nDigite: ");
        scanf("%d", &operacao);

        if(operacao == 1) {
            printf("Digite o numero A: ");
            scanf("%d", &numeroA);
            printf("\nDigite o numero B: ");
            scanf("%d", &numeroB);
            resultado = soma(numeroA, numeroB);
            printf("\nO resultado da soma foi: ");
            printf("%d", resultado);
        } else if (operacao == 2) {
            printf("Digite o numero A: ");
            scanf("%d", &numeroA);
            printf("\nDigite o numero B: ");
            scanf("%d", &numeroB);
            resultado = subtracao(numeroA, numeroB);
            printf("\nO resultado da subtracao foi: ");
            printf("%d", resultado);
        } else if (operacao == 3) {
            printf("Digite o numero A: ");
            scanf("%d", &numeroA);
            printf("\nDigite o numero B: ");
            scanf("%d", &numeroB);
            resultado = multiplicacao(numeroA, numeroB);
            printf("\nO resultado da multiplicacao foi: ");
            printf("%d", resultado);
        } else if (operacao == 4) {
            printf("Digite o numero A: ");
            scanf("%d", &numeroA);
            printf("\nDigite o numero B: ");
            scanf("%d", &numeroB);
            resultado = divisao(numeroA, numeroB);
            printf("\nO resultado da divisao foi: ");
            printf("%d", resultado);
        } else if (operacao == 5) {
            printf("Digite o numero que significa 100 por cento: ");
            scanf("%d", &numeroA);
            printf("\nDigite a porcentagem desejada: ");
            scanf("%d", &numeroB);
            resultado = porcentagem(numeroA, numeroB);
            printf("\nO resultado da porcentagem foi: ");
            printf("%d", resultado);
        }
    }

    return 0;
}
