void loader (int segundos) {
    int i;
    for (i = segundos; i >= 1 ; i--) {
        printf("%d...", i);
        sleep(1);
    }
}

float saqueSimples(float valor, float saldo)
{
    if (valor <= saldo) {
        saldo = saldo - valor;
        printf("Saque realizado.\n");
        printf("Saldo: %.2f\n", saldo);
        return saldo;
    }
    // caso dê erro, retorno -1 que é um valor inválido
    return -1;
}

int main (void)
{
    float valor;
    int conta;
    float chequeEspecial = 100;
    int opcao = 1;
    int i;
    float saldo;
    float resultado;
    float listaClientes[10];
    listaClientes[0] = 100;
    listaClientes[1] = 500;
    listaClientes[2] = 1000;
    listaClientes[3] = 450;
    listaClientes[4] = 600;

    while (opcao > 0)
    {
        opcao = menu();

        if(opcao > 2){
            printf("\nOpcao não suportada!");
        } else if(opcao == 2) {
            printf("\nEstamos trabalhando na funcionalidade de deposito, aguarde!");
        } else if(opcao == 1) {
            //funcionalidade de saque
            conta = identificaContaCliente();
            // informaSaldo(listaClientes, conta);
            informaSaldo(listaClientes[conta]);
            valor = leValorSaque();
            loader(5);
            
            saldo = listaClientes[conta];
            resultado = saqueSimples(valor, saldo);
            if(saldo >= 0) {
                listaClientes[conta] = resultado; // registro o saldo restante na conta do cliente
            } else if (valor <= listaClientes[conta] + chequeEspecial) {
                // caso retorne -1, significa que não foi possível fazer o saque simples, tento o saque com cheque
                //deduzir do saldo
                listaClientes[conta] = listaClientes[conta] - valor;
                //deduzir do cheque especial o valor faltante
                chequeEspecial = chequeEspecial + listaClientes[conta];
                //saldo está negativo
                //zerar o saldo
                listaClientes[conta] = 
                    listaClientes[conta] + 
                    (listaClientes[conta] * -1);
                printf("Saque realizado com cheque especial!\n");
                printf("Saldo: %.2f\n", listaClientes[conta]);
                printf("Cheque especial: %.2f\n", chequeEspecial);
            } else {
                printf("Sem saldo suficiente.\n");
                return 0;
            }
        } else {
            // sair do sistema
            printf("Encerrando...");
            return 0;
        }
    }
}
